Header

Usage

Required

            <head>
  ...
  <script type="text/javascript" href="drop-down.js"></script>
  ...
</head>

'dropDownMenu' Method

API

            dropDownMenu(targetElement, [Menu Template]); // A valid electron menu template

Example

             // Button that is related to the drop down menu
var myButton = document.getElementsByClassName("btn")[0];

dropDownMenu(myButton, [
  {
    label: "Item 1",
    submenu: [
      {
        label: "Sub Item 1.1",
        click: function() {
          console.log("Clicked Sub Item 1.1");
        }
      }
    ]
  },
  {
    label: "Item 2",
    submenu: [
      {
        label: "Sub Item 2.1"
      }
    ]
  }
]);
